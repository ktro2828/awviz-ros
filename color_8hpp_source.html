74inline std::vector<rerun::Color> colormap(const std::vector<float> & values)
75{
76 float min_value = *std::min_element(values.cbegin(), values.cend());
77 float max_value = *std::max_element(values.cbegin(), values.cend());
78
79 std::vector<rerun::Color> colors;
80
81 for (const auto & value : values) {
82 auto idx = static_cast<size_t>(255 * (value - min_value) / (max_value - min_value));
83 colors.emplace_back(rerun::Color(TurboBytes[idx][0], TurboBytes[idx][1], TurboBytes[idx][2]));
84 }
85 return colors;
86}
