142 void set_property(
143 const std::string & topic,
144 const std::shared_ptr<std::unordered_map<std::string, std::string>> entity_roots) override
145 {
146 property_.set_topic(topic);
147 property_.set_entity_roots(entity_roots);
148 set_state(DisplayState::kConfigured);
149 }
150
154 void start() override
155 {
156 // Only transition to running when the display is properly initialized/configured.
157 if (!is_initialized()) {
158 return;
159 }
160 subscribe();
161 set_state(DisplayState::kRunning);
162 }
163
167 void end() override
168 {
169 unsubscribe();
170 set_state(DisplayState::kConfigured);
171 }
172
173 bool is_initialized() const override
174 {
175 return (state() == DisplayState::kConfigured || state() == DisplayState::kRunning) &&
176 property_.is_initialized();
177 }
178
179protected:
180 static constexpr const char * TIMELINE_NAME = "timestamp";
181
186 virtual void subscribe()
187 {
188 if (!is_initialized()) {
189 return;
190 }
191
192 // TODO(ktro2828): QoS setting
193 subscription_ = node_->create_subscription<MsgType>(
194 property_.topic(), rclcpp::SensorDataQoS{},
195 [this](const typename MsgType::ConstSharedPtr msg) { log_message(msg); });
196 };
197
201 virtual void unsubscribe() { subscription_.reset(); }

208 virtual void log_message(typename MsgType::ConstSharedPtr msg) = 0;

210protected:

212 RosTopicProperty property_;
